/**
 * Magical Snowfall Effect for Chrome DevTools Console
 * Injects a full-screen snow overlay onto the current page.
 */
(() => {
  const COUNT = 100; // Number of snowflakes
  const SNOW_COLOR = '#fff';
  
  // Create and inject the necessary CSS
  const style = document.createElement('style');
  style.textContent = `
    .snow-flake {
      position: fixed;
      top: -10px;
      background-color: ${SNOW_COLOR};
      border-radius: 50%;
      pointer-events: none; /* Prevent interference with page interactions */
      z-index: 999999;
      box-shadow: 0 0 5px rgba(255, 255, 255, 0.8);
    }

    @keyframes fall {
      0% {
        transform: translateY(0) translateX(0);
        opacity: 1;
      }
      80% {
        opacity: 1;
      }
      100% {
        transform: translateY(110vh) translateX(50px);
        opacity: 0;
      }
    }
  `;
  document.head.appendChild(style);

  // Helper to generate a random number within a range
  const random = (min, max) => Math.random() * (max - min) + min;

  /**
   * Creates a single snowflake with randomized properties
   */
  const createSnowflake = () => {
    const flake = document.createElement('div');
    flake.className = 'snow-flake';
    
    // Randomize appearance and animation
    const size = random(2, 6);
    const startX = random(0, window.innerWidth);
    const delay = random(0, 10);
    const duration = random(5, 15);
    const opacity = random(0.3, 0.9);

    Object.assign(flake.style, {
      width: `${size}px`,
      height: `${size}px`,
      left: `${startX}px`,
      opacity: opacity,
      animation: `fall ${duration}s linear ${delay}s infinite`
    });

    document.body.appendChild(flake);

    // Clean up flakes if they are removed or the script is rerun
    // (In this specific case, they loop via CSS animation)
  };

  // Generate initial batch of snowflakes
  for (let i = 0; i < COUNT; i++) {
    createSnowflake();
  }

  console.log(`%c✨ The page is now magical! (${COUNT} snowflakes added)`, "color: #00d4ff; font-weight: bold;");
})();
